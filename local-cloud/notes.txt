USE THIS FUNCTION IN TERMINAL TO START UP DynamoDB

function dynamo(){
 cd ~/Desktop/repositories/aws-dropbox/local-cloud
 java -Djava.library.path=./DynamoDBLocal_lib/ -jar DynamoDBLocal.jar
}

THEN YOU CAN JUST CALL IT IN THE TERMINAL WITH 'dynamo'
